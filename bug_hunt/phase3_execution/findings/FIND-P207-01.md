```python
from flowmap.engine import DensityEngine
from flowmap.core import BookLevel, BBO
import numpy as np

e = DensityEngine(history_width=32, max_levels=50, decay=0.92)
e.tick_size = 0.1
e._tick_size_detected = True
e.ticks_per_row = 1
e.vertical_smoothing = 0.0
e.resize(40, 32)
mid = 100.0
# Bid wall ABOVE mid (crossed residual / stale)

    BookLevel(price=100.2, bid_size=5000.0, ask_size=0.0),  # mid+2 ticks
    BookLevel(price=99.8, bid_size=100.0, ask_size=0.0),
    BookLevel(price=100.1, bid_size=0.0, ask_size=100.0),

bbo = BBO(0, "T", 99.95, 100.05, 1.0, 1.0)
e.push_snapshot(levels, bbo, auto_follow=True, vis_rows=40)
buf = e.get_buffer()
col = buf[:, -1, :]
# Row for price 100.2 in render ticks
row = (buf.shape[0]//2) - int(round(100.2 / e.render_tick_size)) + e.center_price_ticks
# EXPECT: bid-colored pixel; ACTUAL: BG (no bid paint above mid)
assert not np.array_equal(col[row], [0, 0, 0, 255])  # FAILS today
```

Side-of-book sizes paint regardless of mid half-plane (or documented mid-mask product policy with UI note). Bid liquidity resting above mid remains visible.
